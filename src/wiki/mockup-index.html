Technical Implementation Plan

1. Authentication Check 

// Get current user
const user = await supabase.getCurrentUser();

// Check if user owns this content
const isOwner = content.author_id === user?.id ||
                content.organizer_id === user?.id ||
                content.created_by === user?.id;

2. Conditional Rendering 

// Show edit/delete buttons only for owner
${isOwner ? `
  <a href="wiki-editor.html?slug=${slug}&type=${type}" class="btn btn-primary">
    <i class="fas fa-edit"></i> Edit
  </a>
  <button onclick="deleteContent('${id}')" class="btn btn-danger">
    <i class="fas fa-trash"></i> Delete
  </button>
` : ''}

3. Delete Confirmation 

async function deleteContent(id, type) {
  const confirmed = confirm('Are you sure you want to delete this?');
  if (!confirmed) return;

  const table = type === 'guide' ? 'wiki_guides' :
                type === 'event' ? 'wiki_events' :
                'wiki_locations';

  await supabase.delete(table, id);
  location.reload(); // Refresh the page
}
